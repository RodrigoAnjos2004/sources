    <h2>Login</h2>
    <input type="text" id="login-username" placeholder="Nome de Usuário">
    <input type="password" id="login-password" placeholder="Senha">
    <button onclick="attemptLogin()">Login</button>
  </div>

  <script>
    const registeredUsers = [
      { username: 'usuario1', cpf: '123456789', email: 'usuario1@example.com', password: 'senha1' },
      // Adicione mais usuários registrados aqui
    ];

    function attemptLogin() {
      const username = document.getElementById('login-username').value;
      const password = document.getElementById('login-password').value;

      const user = registeredUsers.find(u => u.username === username && u.password === password);

      if (user) {
        alert(`Bem-vindo, ${username}!`);
      } else {
        alert('Usuário não registrado ou senha incorreta.');
      }
    }
  </script>
